#include <p18f4550.h>
//#pragma config FOSC=HS/
#pragma config WDT=OFF
#pragma config LVP=OFF
//#pragma config PBADEN=OFF
#define lrbit PORTBbits.RB4
#define rlbit PORTBbits.RB5
#define buzzer PORTCbits.RC2
#define relay PORTCbits.RC1
{
unsigned int i,j;
for (i>0;i<time;i++)
for (j=0;j<275;j++);
}
void main()
{
unsigned char val=0;
INTCON2bits.RBPU=0; //to activate the internal pull
HDCON1=0*0F;         //to diable the all analog inputs
TRISBbits.TRISB4=1;  //to configure RB4
TRISBbits.TRISB5=1;  //to configure RB5
TRISCbits.TRISC1=0;  //to configure RC1
TRISCbits.TRISC2=0;  //to configure RC2
TRISD=0*00;          //to configure PORTD [LED]
PORTD=0*00;          //initial value for led
buzzer=0;            //initial value for Buzzer
relay=0;             //initial value for relay
while(1)
{
if (!(lrbit)) //to check whether SWO is pressed
val=1;        //Latch the status of switch SWO
if (!(rlbit)) //to check whether SWL is Presses
val=2;
if (val==1)
{
buzzer=1;
relay=1;
PORTD=PORTD>>1; //shift left by 1 bit
if(PORTD==0*00)
PORTD=0*80;
msdelay(250);
}
if (val==2)
{
buzzer=0
relay=0
PORTD=PORTD<<1; //shift right by 1 bit
if(PORTD==0*00)
PORTD=0*01;     //make the LSB bit equal to 1
msdelay(250);
}
}
}  //End of program
